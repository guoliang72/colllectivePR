#include <stdio.h>
#include <string.h>

int main()
{
    int i,j,n;
    int a;
	int b[50];
	scanf("%d",&n);
    for(i=0;i<n;i=i+1)
    {
        scanf("%d",&a);
		// 少特判
        j=0;
        while(a>0)
        {
            b[j]=a%3;
            a = a/3;
			j=j+1;
        }
        j=j-1;
        while(j>=0) {
            printf("%d",b[j]);
			j=j-1;
		}
        printf("\n");
    }

    return 0;
}
