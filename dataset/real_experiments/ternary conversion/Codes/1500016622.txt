#include<stdio.h>
#include<math.h>
int main()
{
	int n,i,j,k,m;
	int a[100],t;
	scanf("%d",&n);
	for(i=0;i<n;i=i+1)
	{
		t=0;
		scanf("%d",&m);		// 缺少m=0的判断
		for(j=0;m>0;j=j+1)
		{
			a[j]=m%3;
			m=m/3;
			t=t+1;
		}
		for(k=t-1;k>=0;k=k-1)
		{
			printf("%d",a[k]);
		}
		printf("\n");
	}
	return 0;
}
