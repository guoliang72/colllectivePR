#include <stdio.h>
#include <string.h>
#include <math.h>
int main() {
    int n;
	int a,i,t;
        int count;
        int ans[10000]={0};
    scanf("%d",&n);
    for(t=0;t<n;t++){
        count=0;
        scanf("%d",&a);
        if(a==0){
            printf("0\n");return 0;			// 错误的return;
        }
        for(count=0;a>0;count=count+1){
            ans[count]=a%3;
            a=a/3;
        }
        for(i=count-1;i>=0;i=i-1)
            printf("%d",ans[i]);
        printf("\n");
    }
}
