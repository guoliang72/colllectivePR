#include<stdio.h>
#include<math.h>
int main(){
	int n,c,i,j;
	int num;
	scanf("%d",&n);
	for(c=0;c<n;c=c+1){
		
		scanf("%d",&num);
		int numin3[1000]={0}; // 少一个num==0判断
		for(i=0;num!=0;i=i+1){
			numin3[i]=num%3;
			num=num/3;
		}
		for(j=i-1;j>=0;j=j-1)
			printf("%d",numin3[j]);
		printf("\n");
	}
	return 0;
}
