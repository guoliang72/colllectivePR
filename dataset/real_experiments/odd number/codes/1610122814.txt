#include<stdio.h>
int main(){
    int sz[100], zs[100]={0};
    int n, i, j, a;
    scanf("%d",&n);
    for(i=0;i<n;i=i+1){
        scanf("%d",&sz[i]);
    }
    for(i=0;i<n;i=i+1){
        a=sz[i];
        zs[a]=zs[a]+1;
    }
    for(i=0;i<n;i=i+1){			// 应该i<100
        if(zs[i]%2!=0){
            printf("%d",i);
            break;
        }
    }
    return 0;

}
