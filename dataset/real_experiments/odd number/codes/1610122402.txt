#include<stdio.h>
int main(){
    int n,i,j,c;
    scanf("%d",&n);
    int a[n],b[101];
    for(i=0;i<n;i=i+1){
        scanf("%d",&a[i]);
    }
    for(j=0;j<101;j=j+1){
            b[j]=0;
        for(i=0;i<n;i=i+1){
            if(a[i]==j){
                b[j]=b[j]+1;
            }
        }
    }
    for(i=0;i<101;i=i+1){
        if((b[i]/2)!=0){		// 符号错误应为%
            c=i;
        }
    }
    printf("%d",c);
    return 0;
}
