#include <stdio.h>
int main(){
	int n,i;
	scanf("%d",&n);
	double a[1000],sum;
	sum = 0;
	for(i=0;i<n;i=i+1){
		scanf("%lf",&a[i]);
	
		if(a[i]<100){
			sum=sum+a[i];
		}
		if(a[i]>=100&&a[i]<200){
			sum=sum+a[i]*0.9;		// 式子错误
		}
		if(a[i]>=200&&a[i]<500){
			sum=sum+a[i]*0.7;		//
		}
		if(a[i]>=500){
			sum=sum+a[i]*0.5;		//
		}
	}
	printf("%.2lf",sum);
return 0;
}
