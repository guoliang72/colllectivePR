#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main(){
    int n,i;
    scanf("%d",&n);
    double a[n],s;
    for(i=0;i<n;i=i+1){
        scanf("%lf",&a[i]);
    if(a[i]<100){
        s=s+0;
    }
    if(a[i]>=100&&a[i]<200){
        s=s+0.1*a[i];
    }
    if(a[i]>=200&&a[i]<500){
        s=s+0.3*a[i];
    }
    if(a[i]>=500){  
        s=s+0.5*a[i];
    }
    
    }
    printf(".2lf",s);			// 输出格式错误
   
    return 0;
}
