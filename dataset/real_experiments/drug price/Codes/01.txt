#include<stdio.h>
int main(){
	int n,i;
	double sum;
	sum = 0;
	double a;
	scanf("%d",&n);
	for(i=0;i<n;i=i+1){       
		scanf("%lf",&a);
		if(a<100){
			a=0;
		}else if(a>=100&&a<200){
			a=0.9*a;				// 常数错
		}else if(a>=200&&a<500){
			a=0.7*a;				//
		}else{
			a=0.5*a;				//
		}
		sum = sum + a;
	}
	printf("%.2lf",sum);
	return 0;
}
