#include <stdio.h>
int main(){
int n,i;
scanf("%d",&n);
double a[n],b,sum;
sum = 0;
for(i=0;i<n;i=i+1){
	scanf("%d",&a[i]);		// 输入格式错误
	if(a[i]<100){
		b=0;
	}
	else if(a[i]<200){
		b=0.1*a[i];
	}
	else if(a[i]<500){
		b=0.3*a[i];
	}
	else{
		b=0.5*a[i];
	}
	sum = sum + b;
}
printf("%.2lf",sum);
return 0;
}
